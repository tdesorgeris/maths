\card{1}{
\type{definition}
\question{
Donner la definition d'une suite convergente.
}
\answer{
La suite $(u_n)$ converge vers $\ell \in \R$ si, pour tout $\eps > 0$, il existe
$N \in \N$ tel que, pour tout $n \geq N$, on ait $\abs{u_n-\ell} \leq \eps$.
}
}
