\questioncours{1}{
Enoncer le theoreme des valeurs intermediaires.
}

\exercice{1}{
\difficulty{1}
\title{Mon premier exercice}
\statement{
Soit $f : \R \to \R$ definie par $f(x)=x^2-2$.
Montrer que l'equation $f(x)=0$ admet une solution dans $[1,2]$.
}
\answer{
On remarque que $f$ est continue sur $[1,2]$, avec $f(1)=-1$ et $f(2)=2$.
On applique alors le theoreme des valeurs intermediaires.
}
}
